Funo para Benchmark - R
trials <- function(action, iterations) {
tempos <- numeric()
for (i in 1:iterations) {
tempo1 <- Sys.time()
action()
tempo2 <- Sys.time()
tempos <- c(tempos,as.numeric(tempo2-tempo1))
}
print(mean(tempos))
print(sd(tempos))
}

Arquivo no formato Parquet
Os arquivos Parquet so um formato de armazenamento de dados colunar que
oferece alta eficincia para trabalhar com grandes volumes de informaes, j
que apenas as colunas relevantes para uma determinada anlise precisam ser
lidas do disco, em vez de todo o conjunto de dados.
Diferente dos formatos tradicionais como CSV que armazenam dados linha por
linha, o Parquet organiza as informaes por colunas, o que significa que todos
os valores de uma mesma coluna so armazenados juntos.

Importao em Chunks - Python
import pandas
def importacao_chunks_python():
#definindo tamanho do chunk, criando variveis para armazenar valores
reviews_positivo = 0
total_reviews = 0
chunks = pandas.read_csv("imdb.csv", chunksize=2000)
#loop que importa cada chunk
for chunk in chunks:
reviews_positivo += (chunk["sentiment"] == "positive").sum()
total_reviews += len(chunk)
print(f"porcentagem de reviews positivos: {100*reviews_positivo/total_reviews}%")
trials(importacao_chunks_python,10)

Importao em Chunks - Julia
using CSV
using DataFrames
function importacao_chunks_julia()
reviews_positivo = 0
total_reviews = 0
chunk_size = 2000
buffer = DataFrame()
#l linha por linha, acumulando em chunks de 2000 (procedimento "manual", comparando com pacotes R ou Python)
for row in CSV.File("imdb.csv")
push!(buffer, row)
if nrow(buffer) == chunk_size
reviews_positivo += sum(buffer.sentiment .== "positive")
total_reviews += chunk_size
empty!(buffer) #limpa para o prximo chunk
end
end
porc = 100 * reviews_positivo / total_reviews
println("porcentagem de reviews positivos: $(round(porc, digits=2))%")
end
